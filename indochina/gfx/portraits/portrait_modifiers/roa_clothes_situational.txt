			}

		outfit_tags = { china_commoner_outfit }
		weight = {
			base = 0
			modifier = {

				portrait_chinese_clothing_trigger = yes
			}
			modifier = {
				add = 20
				highest_held_title_tier < tier_county
				NOR = {
					AND = {
						exists = primary_spouse
						primary_spouse = {
							highest_held_title_tier > tier_county
						}
					}
					AND = {
						has_government = republic_government
						primary_title.tier > 1
					}
					has_government = theocracy_government
				}
				portrait_chinese_clothing_trigger = yes
			}
		}
	}

							OR = {

							portrait_chinese_clothing_trigger = yes
						}
					}
				}
			}
			modifier = {
				add = 20
				has_government = anarchist_government
				NOT = { portrait_era1_trigger = yes } #current_date >= 960.1.1
				portrait_chinese_clothing_trigger = yes
			}
			modifier = {
				add = 180
				exists = this
				is_adult = yes
				NOT = { portrait_era1_trigger = yes } #current_date >= 960.1.1
				OR = {
					has_government = chinese_government
					AND = {
				        exists = liege
				        liege = {
							has_title = title:e_celestia_china
							has_government = chinese_government
				        }
				    }
				}
				portrait_chinese_wear_helmet_trigger = no
			}
			modifier = {
				add = -180
				exists = this
				NOR = {
					has_government = chinese_government
				}
			}
		}
