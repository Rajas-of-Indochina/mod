on_birthday = {
   on_actions = { thamade_action }
}

thamade_action = {
	trigger = {
		OR  {
			is_ai = yes
			is_courtier	= yes
		}
 culture = { has_cultural_pillar = heritage_caucasian }
 age > 50
 any_character_trait = {
   count >= 1
   trait_is_virtue = yes
}
 any_character_trait = {
   count = 0
   trait_is_sinful = yes
}
	}
	effect = {
add_trait = trait_thamade
	}
}
