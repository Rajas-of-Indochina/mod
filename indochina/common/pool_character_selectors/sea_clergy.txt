pool_court_chaplain = {
	valid_character = {
		# Capable adults
		is_capable_adult = yes
		is_imprisoned = no
		is_married = no
		is_betrothed = no

		# In minimum good standing with faith
		NOT = { has_trait = excommunicated }

		# Correct gender
		OR = {
			AND = {
				scope:base.faith = { trait_is_virtue = transgender }
				has_trait = transgender
			}
			scope:base.faith = { has_allowed_gender_for_clergy = root }
		}
		
		# No close family members for temporal religions unless they have relevant traits
		trigger_if = {
			limit = {
				scope:base.faith = {
					NOT = { has_doctrine_parameter = lay_clergy }
				}
			}
			OR = {
				NOT = { is_close_family_of = scope:base }
				has_trait = devoted
				has_trait = saint
			}
		}
	}
	
	character_score = {
		base = 1
		
		modifier = {
			add = 100
			has_same_culture_as = scope:base
		}
		modifier = {
			add = 15
			culture = { has_same_culture_heritage = scope:base.culture }
		}
		modifier = {
			add = 10
			culture = { has_same_culture_language = scope:base.culture }
		}
		modifier = {
			add = 5
			culture = { has_same_culture_ethos = scope:base.culture }
		}

		modifier = {
			factor = 1.5
			age > 40
		}
		modifier = {
			factor = 3
			scope:base.faith = { has_preferred_gender_for_clergy = root }
		}
	}
	
	config = {
		background = clergy
		age = { 25 55 }
	}
	
	selection_count = 25
	
	pool = faith
	gender = clergy
}

		is_imprisoned = no
		is_married = no
		is_betrothed = no

		OR = {
			AND = {
				scope:base.faith = { trait_is_virtue = transgender }

			}
			scope:base.faith = { has_allowed_gender_for_clergy = root }
		}
	}

	character_score = {
		base = 1

		modifier = {
			add = 1000
			is_courtier_of = scope:base
		}
		compare_modifier = {
			value = martial
		}
		modifier = {
			dynasty.dynasty_prestige_level >= 2
			add = {
				value = dynasty.dynasty_prestige_level
				multiply = 20
				subtract = 20
			}
		}
		modifier = {
		piety_level >= 2
			add = {
				value = piety_level
				multiply = 5
			}
		}
		modifier = {
			add = 100
			has_same_culture_as = scope:base
		}
		modifier = {
			add = 15
			culture = { has_same_culture_heritage = scope:base.culture }
		}
		modifier = {
			add = 10
			culture = { has_same_culture_language = scope:base.culture }
		}
		modifier = {
			add = 5
			culture = { has_same_culture_ethos = scope:base.culture }
		}
		modifier = {
			add = 50
			age > 35
		}
		modifier = {
			add = 100
			scope:base.faith = { has_preferred_gender_for_clergy = root }
		}
	}

	config = {
		background = commander
		age = { 25 45 }
	}

	selection_count = 25

	pool = faith

}
