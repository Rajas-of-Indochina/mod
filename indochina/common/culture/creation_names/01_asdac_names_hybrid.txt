upper_saxon_hybrid = {
	trigger = {
		OR = {
			AND = {
				scope:culture = culture:saxon
				scope:other_culture = culture:sorbian
			}
			AND = {
				scope:culture = culture:sorbian
				scope:other_culture = culture:saxon
			}
		}
		any_sub_realm_county = {
			OR = {
				culture = scope:culture
				culture = scope:other_culture
			}
		}
	}
	hybrid = yes
}
litvin_hybrid = {
	trigger = {
		OR = {
			AND = {
				scope:culture = culture:lithuanian
				scope:other_culture = { has_cultural_pillar = heritage_east_slavic }
			}
			AND = {
				scope:culture = { has_cultural_pillar = heritage_east_slavic }
				scope:other_culture = culture:lithuanian
			}
		}
		any_sub_realm_county = {
			OR = {
				culture = scope:culture
				culture = scope:other_culture
			}
		}
	}
	hybrid = yes
}
