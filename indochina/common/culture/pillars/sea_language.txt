language_yi = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_yi
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_yi }
			multiply = 10
		}
	}

	color = yi
}
