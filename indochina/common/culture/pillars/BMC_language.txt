language_dagestani = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_dagestani
		}
	}
	parameters = {
		language_group_indo_iranian = yes
		language_family_indo_european = yes 
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_dagestani }
			multiply = 10
		}
		else_if = {
			limit = { has_cultural_parameter = language_group_indo_iranian }
			multiply = 5
			}
		else_if = {
			limit = { has_cultural_parameter = language_family_indo_european }
			multiply = 2.5
		}
	}
	
	color = { 109 186 124 }
}
