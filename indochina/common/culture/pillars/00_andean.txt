language_puquina = {
	type = language
	is_shown = {
		language_is_shown_trigger = language_puquina
	}
	parameters = {
		language_group_andean = yes
		language_family_south_american = yes
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_puquina }
			multiply = same_language_choice_factor
		}
		else_if = {
		 	limit = { has_cultural_parameter = language_group_andean }
		 	multiply = same_language_group_choice_factor
		}
		else_if = {
			limit = { has_cultural_parameter = language_family_south_american }
			multiply = same_language_family_choice_factor
		}
	}
}
