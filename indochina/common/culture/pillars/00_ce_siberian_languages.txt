language_samoyed = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_samoyed
		}
	}
	parameters = {
		language_group_ugric = yes
		language_family_uralic = yes
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_samoyed }
			multiply = same_language_choice_factor
		}
		else_if = {
			limit = { has_cultural_parameter = language_group_ugric }
			multiply = same_language_group_choice_factor
		}
		else_if = {
			limit = { has_cultural_parameter = language_family_uralic }
			multiply = same_language_family_choice_factor
		}
	}

	color = samoyed
}
