
county_has_minority_faith_trigger = {
	OR = {
		AND = {
			has_variable_list = faith_minorities_small
			variable_list_size = {
	            name = faith_minorities_small
	            value > 0
	        }
		}
		AND = {
			has_variable_list = faith_minorities_large
			variable_list_size = {
	            name = faith_minorities_large
	            value > 0
	        }
		}
	}
}
