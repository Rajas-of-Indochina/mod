building_requirement_cow_metropolis = {
	trigger_if = {
		limit = { has_holding_type = cow_metropolis_holding }
		has_building_or_higher = cow_metropolis_$LEVEL$
	}
	trigger_else = {
		custom_description = {
			text = "BUILDING_REQUIREMENT_COW_METROPOLIS_FAILED"
			always = no
		}
	}
}
