has_imperial_laws = {
	is_landed = yes
	OR = {
		AND = {
			highest_held_title_tier = tier_empire
			has_government = imperial_government 
		}
		AND = {
			has_government = autocratic_government
			is_independent_ruler = yes
		}
	}

}

has_imperial_vassal_laws = {
	OR = {
		has_government = imperial_government 
		has_government = autocratic_government
	}
	is_landed = yes
	any_liege_or_above = {
		has_imperial_laws = yes
	}
	NOT = {
		is_independent_ruler = yes
	}
}
