# $RELIGION$ can be a religion or foreign_faiths
has_involved_large_minority_faith_trigger = {
	any_in_list = {
		variable = faith_minorities_large
		struggle:indian_struggle = {is_faith_involved_in_struggle = PREV}
		trigger_if = {
			limit = {
				OR = {
					$RELIGION$ = religion:buddhism_religion
					$RELIGION$ = religion:hinduism_religion
					$RELIGION$ = religion:jainism_religion
					$RELIGION$ = religion:islam_religion
					$RELIGION$ = religion:indian_religion
				}
			}
			religion = $RELIGION$
		}
		trigger_else = {
			NOR = {
				religion = religion:buddhism_religion
				religion = religion:hinduism_religion
				religion = religion:jainism_religion
				religion = religion:islam_religion
				religion = religion:indian_religion
			}
		}
	}
}
