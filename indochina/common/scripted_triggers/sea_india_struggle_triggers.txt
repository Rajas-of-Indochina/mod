india_struggle_dominant_phase_trigger = {
	exists = struggle:indian_struggle
	OR = {
		AND = {
			religion = religion:buddhism_religion
			any_character_struggle = {
				is_struggle_phase = struggle_india_buddhist_dominance
			}
		}
		AND = {
			religion = religion:hinduism_religion
			any_character_struggle = {
				is_struggle_phase = struggle_india_hindu_dominance
			}
		}
		AND = {
			religion = religion:jainism_religion
			any_character_struggle = {
				is_struggle_phase = struggle_india_jain_dominance
			}
		}
		AND = {
			religion = religion:indian_religion
			any_character_struggle = {
				is_struggle_phase = struggle_india_indian_animist_dominance
			}
		}
		AND = {
			religion = religion:islam_religion
			any_character_struggle = {
				is_struggle_phase = struggle_india_muslim_dominance
			}
		}
		AND = {
			NOR = {
				religion = religion:buddhism_religion
				religion = religion:hinduism_religion
				religion = religion:jainism_religion
				religion = religion:indian_religion
				religion = religion:islam_religion
			}
			any_character_struggle = {
				is_struggle_phase = struggle_india_foreign_dominance
			}
		}
	}
}
