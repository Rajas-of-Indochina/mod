
##################################################
# General Triggers

tibetan_struggle_character_any_involvement_tibetan_struggle_trigger = {
	OR = {
		tibetan_struggle_character_involved_in_struggle_trigger = yes
		tibetan_struggle_character_interloper_in_struggle_trigger = yes
	}
}

tibetan_struggle_character_involved_in_struggle_trigger = {
	any_character_struggle = {
		involvement = involved
		is_struggle_type = tibetan_struggle
	}
}

tibetan_struggle_character_interloper_in_struggle_trigger = {
	any_character_struggle = {
		involvement = interloper
		is_struggle_type = tibetan_struggle
	}
}

tibetan_struggle_character_uninvolved_in_struggle_trigger = {
	exists = struggle:tibetan_struggle
	tibetan_struggle_character_interloper_in_struggle_trigger = no
	tibetan_struggle_character_involved_in_struggle_trigger = no
}
