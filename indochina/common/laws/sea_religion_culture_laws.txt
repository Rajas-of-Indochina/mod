@minority_law_cooldown_years = 10

religious_minority_status_law = {
	default = religious_minority_status_2
	cumulative = yes
	flag = religion_culture_law
	flag = religion_law

	# Laws only apply to astray and lower faiths

	religious_minority_status_1 = {
		## Accepted
		# Bonus opinion
		# Minorities may grow
		# May not promote faith

		modifier = {
			different_faith_opinion = 10
		}

		can_have = { NOT = { has_government = anarchist_government } }
		can_keep = { NOT = { has_government = anarchist_government } }

		can_pass = {
			OR = {
				has_realm_law = religious_minority_status_2
			}
			custom_description = {
				subject = root
				text = "has_minority_law_cooldown"
				NOT = { has_variable = minority_law_cooldown }
			}
		}

		pass_cost = {
			prestige = increase_crown_authority_prestige_cost
		}

		on_pass = {
			set_variable = {
				name = minority_law_cooldown
				years = @minority_law_cooldown_years
			}
		}
	}

	religious_minority_status_2 = {
		## Tolerated
		# Default
		# May convert counties

		flag = may_convert_faith

		modifier = {
			different_faith_opinion = -10
		}

		should_start_with = { always = yes }

		can_have = { NOT = { has_government = anarchist_government } }
		can_keep = { NOT = { has_government = anarchist_government } }

		can_pass = {
			OR = {
				has_realm_law = religious_minority_status_1
				has_realm_law = religious_minority_status_3
			}
			custom_description = {
				subject = root
				text = "has_minority_law_cooldown"
				NOT = { has_variable = minority_law_cooldown }
			}
		}

		pass_cost = {
			prestige = increase_crown_authority_prestige_cost
		}

		on_pass = {
			set_variable = {
				name = minority_law_cooldown
				years = @minority_law_cooldown_years
			}
		}
	}

	religious_minority_status_3 = {
		## Suspicious (?)
		# Minorities can be removed
		# Less opinion
		# Unlocks expel decision

		modifier = {
			different_faith_opinion = -10
		}

		flag = unlock_expel_religious_minorities

		can_have = { NOT = { has_government = anarchist_government } }
		can_keep = { NOT = { has_government = anarchist_government } }

		can_pass = {
			OR = {
				has_realm_law = religious_minority_status_2
				has_realm_law = religious_minority_status_4
			}
			custom_description = {
				subject = root
				text = "has_minority_law_cooldown"
				NOT = { has_variable = minority_law_cooldown }
			}
		}

		pass_cost = {
			prestige = increase_crown_authority_prestige_cost
		}

		on_pass = {
			set_variable = {
				name = minority_law_cooldown
				years = @minority_law_cooldown_years
			}
		}
	}

	religious_minority_status_4 = {
		## Targeted (Purge?)
		# Large negative to opinion
		# Unlocks purge decision

		modifier = {
			different_faith_opinion = -20
		}

		flag = unlock_purge_religious_minorities

		can_have = { NOT = { has_government = anarchist_government } }
		can_keep = { NOT = { has_government = anarchist_government } }

		can_pass = {
			OR = {
				has_realm_law = religious_minority_status_3
			}
			custom_description = {
				subject = root
				text = "has_minority_law_cooldown"
				NOT = { has_variable = minority_law_cooldown }
			}
		}

		pass_cost = {
			prestige = increase_crown_authority_prestige_cost
		}

		on_pass = {
			set_variable = {
				name = minority_law_cooldown
				years = @minority_law_cooldown_years
			}
		}
	}
}
