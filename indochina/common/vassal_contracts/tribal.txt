tribal_government_obligations = {
	obligation_levels = {
		default = {
			levies = {
				value = 0
				max = 0.5
				scope:liege = {
					add = {
						value = 0.1
						multiply = prestige_level
					}
				}
				scope:vassal = {					
					if = {
						limit = {
							AND = {
								scope:liege = { # Dominant faith co-religionist vassals contribute more
									religion = scope:vassal.religion
									#india_struggle_dominant_phase_trigger = yes
								}
								any_character_struggle = {
									is_struggle_type = indian_struggle
								}
							}
						}
						add = {
							value = 0.05
						}
					}
				}
			}
			tax = {
				value = 0
				max = 0.25
				scope:liege = {
					add = {
						value = 0.05
						multiply = prestige_level
					}
				}
				scope:vassal = {					
					if = {
						limit = {
							AND = {
								scope:liege = { # Dominant faith co-religionist vassals contribute more
									religion = scope:vassal.religion
									#india_struggle_dominant_phase_trigger = yes
								}
								any_character_struggle = {
									is_struggle_type = indian_struggle
								}
							}
						}
						add = {
							value = 0.05
						}
					}
				}
			}
		}
	}
}
