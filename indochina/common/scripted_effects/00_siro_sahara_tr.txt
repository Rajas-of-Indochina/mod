siro_e_yearly_revaluate_sh = {
	change_global_variable = { ## keep this value untill modifier changes, if this value reachs interval, modifier will be changed.
		name = siro_g_sh_val_c_keep
		add = global_var:siro_g_sh_val_c
	}	
	if = {
		limit = {
			OR = {
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value < 20
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							compare_value >= 4
							compare_value <= -4

				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 20
						compare_value < 29
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							compare_value >= 3
							compare_value <= -3
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 29
						compare_value < 35
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							compare_value >= 2
							compare_value <= -2
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 35
						compare_value < 39
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							compare_value >= 1
							compare_value <= -1
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 39
						compare_value < 69
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							compare_value >= 2
							compare_value <= -2
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 69
						compare_value < 111
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							compare_value >= 3
							compare_value <= -3
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 111
						compare_value < 171
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							compare_value >= 4
							compare_value <= -4
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 171
						compare_value < 246
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							compare_value >= 5
							compare_value <= -5
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 246
						compare_value < 336
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							compare_value >= 6
							compare_value <= -6
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						compare_value >= 336
						compare_value < 413
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							compare_value >= 7
							compare_value <= -7
						}
					}
				}
			}
		}
		change_global_variable = {
			name = siro_g_sh_val
			add = global_var:siro_g_sh_val_c
		}	
		siro_e_sh_remove_modifier = yes
		siro_e_yearly_global_sh_modifier = yes
		
		set_global_variable = {
			name = siro_g_sh_val_c_keep
			value = 0
		}
		set_global_variable = {
			name = siro_g_sh_val_c
			value = 0
		}
	}
	else = {
		change_global_variable = {
			name = siro_g_sh_val
			add = global_var:siro_g_sh_val_c
		}	
		set_global_variable = {
			name = siro_g_sh_val_c
			value = 0
		}
	}
}
