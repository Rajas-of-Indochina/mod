trigger_india_struggle_catalyst = {
	if = {
		limit = {
			NOR = {
				$RELIGION$ = religion:religion_buddhism
				$RELIGION$ = religion:religion_hinduism
				$RELIGION$ = religion:religion_jainism
				$RELIGION$ = religion:religion_islam
				$RELIGION$ = religion:religion_indian_religion
			}
		}
		every_character_struggle = {
			activate_struggle_catalyst = {
				catalyst = $CATALYST$_$RELIGION$
				character = $CHARACTER$
			}
		}
	}
	every_character_struggle = {
		activate_struggle_catalyst = {
			catalyst = $CATALYST$_foreign_religion
			character = $CHARACTER$
		}
	}
}
