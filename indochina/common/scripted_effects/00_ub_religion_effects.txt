ub_set_bhakti_effect = {
	if = {
		limit = { religion = religion:hellenism_religion}
		if = {
			limit = {
				has_character_modifier = bhakti_hellenic_zeus
			}
			remove_character_modifier = bhakti_hellenic_zeus
			add_piety = massive_piety_loss
		}
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_athena
			}
			remove_character_modifier = bhakti_hellenic_athena
			add_piety = massive_piety_loss
		} 
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_artemis
			}
			remove_character_modifier = bhakti_hellenic_artemis
			add_piety = massive_piety_loss
		}
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_poseidon
			}
			remove_character_modifier = bhakti_hellenic_poseidon
			add_piety = massive_piety_loss
		}
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_roman_saturnus
			}
			remove_character_modifier = bhakti_hellenic_roman_saturnus
			add_piety = massive_piety_loss
		}
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_roman_mithra
			}
			remove_character_modifier = bhakti_hellenic_roman_mithra
			add_piety = massive_piety_loss
		}
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_roman_janus
			}
			remove_character_modifier = bhakti_hellenic_roman_janus
			add_piety = massive_piety_loss
		}
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_roman_pluto
			}
			remove_character_modifier = bhakti_hellenic_roman_pluto
			add_piety = massive_piety_loss
		}
		else_if = {
			limit = {
				has_character_modifier = bhakti_hellenic_roman_bacchus
			}
			remove_character_modifier = bhakti_hellenic_roman_bacchus
			add_piety = massive_piety_loss
		}
	}
	else = {
		# If no current bhakti, cost is reduced.
		add_piety = medium_piety_loss
	}

	add_character_modifier = $BHAKTI$
}
