#Set Shiraz(Fars) as Capital of Persia

ub_set_capital_fars_decision = {
	picture = "gfx/interface/illustrations/decisions/decisions_persepolis_images.dds"

	desc = ub_set_capital_fars_decision_desc
	selection_tooltip = ub_set_capital_fars_decision_tooltip
	major = yes

	ai_check_interval = 12

	is_shown = {
		is_ruler = yes
		is_landed = yes
		has_title = title:e_persia
		NOT = { capital_county = { this = title:c_shiraz } }
		OR = {
			title:c_shiraz.holder = { 	
				any_liege_or_above = { this = root }
			}
			title:c_shiraz.holder = { this = root }
		}
		title:b_shiraz-farsi = {
			title_province = {
				has_building = ub_persepolis_02
			}
		}
	}

	is_valid_showing_failures_only = {
		is_capable_adult = yes
		is_imprisoned = no
		is_independent_ruler = yes
	}

	effect = {
		if = { #Usurp if not held personally.
			limit = {
				NOT = { title:c_shiraz.holder = { this = root } }
			}
			create_title_and_vassal_change = {
				type = returned
				save_scope_as = change
				add_claim_on_loss = no
			}
			title:c_shiraz = {
				change_title_holder = {
					holder = root
					change = scope:change
				}
			}
			resolve_title_and_vassal_change = scope:change

		}
		if = { #if Shiraz isn't the de jure captital of e_persia, make it so
			limit = {
				NOT = { title:c_shiraz = title:e_persia.title_capital_county }
			}
			title:e_persia = {
				set_capital_county = title:c_shiraz
			}
		}
		set_realm_capital = title:c_shiraz
		play_music_cue = "mx_cue_positive_effect"
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = { #Do it always, for coherence.
		base = 100
	}
}
