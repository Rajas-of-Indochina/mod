change_from_city_state = {
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	desc = change_from_city_state_desc
	selection_tooltip = change_from_city_state_tooltip
	major = yes
	ai_check_interval = 500

	is_shown = {
		is_ruler = yes
		is_landed = yes
		has_government = city_state_government

		OR = {
			faith = { has_doctrine = doctrine_right_to_rule_divine }
			faith = { has_doctrine = doctrine_right_to_rule_rajadharma }
			faith = { has_doctrine = doctrine_right_to_rule_material }
		}

		realm_size >= 15
	}

	is_valid = {
		is_independent_ruler = yes
		prestige_level >= 3
	}

	effect = {
		if = {
			limit = {
				faith = { has_doctrine = doctrine_right_to_rule_divine }
			}
			change_government = feudal_government

			every_vassal_or_below = {
				limit = {
					has_government = city_state_government
					faith = { has_doctrine = doctrine_right_to_rule_divine }
				}
				change_government = feudal_government
			}
		}
		else_if = {
			limit = {
				faith = { has_doctrine = doctrine_right_to_rule_rajadharma }
			}
			change_government = mandala_government

			every_vassal_or_below = {
				limit = {
					has_government = city_state_government
					faith = { has_doctrine = doctrine_right_to_rule_rajadharma }
				}
				change_government = mandala_government
			}
		}
		else_if = {
			limit = {
				faith = { has_doctrine = doctrine_right_to_rule_material }
			}
			change_government = clan_government

			every_vassal_or_below = {
				limit = {
					has_government = city_state_government
					faith = { has_doctrine = doctrine_right_to_rule_material }
				}
				change_government = clan_government
			}
		}

		hidden_effect = {
			if = {
				limit = {
					OR = {
						has_government = feudal_government
						has_government = clan_government
					}
					NOT = { has_realm_law = crown_authority_0 }
				}
				add_realm_law_skip_effects = crown_authority_0
			}
			if = {
				limit = {
					has_government = mandala_government
					NOT = { has_realm_law = mandala_authority_0 }
				}
				add_realm_law_skip_effects = mandala_authority_0
			}
		}
	}

	ai_potential = {
		always = no
	}

	cost = {
		gold =  150
		prestige = 200
	}
}

change_to_caliphate = {
	picture = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
	desc = change_to_caliphate_desc
	selection_tooltip = change_to_caliphate_tooltip
	major = yes
	ai_check_interval = 120

	is_shown = {
		is_ruler = yes
		is_landed = yes
		has_religion = religion:islam_religion
		has_government = muslim_government
		NOT = {
			has_government = caliphate_government
		}
	}

	is_valid = {
		prestige_level >= 3
	}

	effect = {
		change_government = caliphate_government
		if = {
			limit = {
				faith = {
					exists = religious_head
				}
			}
			faith = {
				religious_head = {
					add_opinion = {
						target = root
						opinion = -150
						modifier = hate_opinion
					}
				}
			}
		}
		save_scope_as = caliph
		set_variable = {
			name = cal_authority
			value = 10
		}
		set_variable = {
			name = zakah_balance
			value = 0
		}

		every_realm_province = {
			limit = {
				barony = { is_holy_site_of = scope:caliph.faith }
			}

			scope:caliph = {
				change_variable = {
					name = cal_authority
					add = 2
				}
			}
		}

		if = {
			limit = {
				NOT = { any_realm_province = { barony = { is_holy_site_of = scope:caliph.faith } } }
			}
			change_variable = {
				name = cal_authority
				subtract = 2
			}
		}

		if = {
			limit = {
				piety_level = max_piety_level
			}
			change_variable = {
				name = cal_authority
				add = 5
			}
		}
		if = {
			limit = {
				piety_level = very_high_piety_level
			}
			change_variable = {
				name = cal_authority
				add = 2
			}
		}
		if = {
			limit = {
				piety_level = high_piety_level
			}
			change_variable = {
				name = cal_authority
				add = 1
			}
		}
		if = {
			limit = {
				piety_level = low_piety_level
			}
			change_variable = {
				name = cal_authority
				subtract = 2
			}
		}

		clamp_variable = {
			name = cal_authority
			max = 100
			min = 0
		}
		hidden_effect = {
			while = {
				count = var:cal_authority
				add_character_modifier = caliph_auth
			}
		}
	}

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 10
	}
	cost = { gold = { value = 500 } prestige = { value = 500 } piety = { value = 700 } }
}
