secret_religion_reveal_faith = {
	major = no
	ai_goal = no
	picture = "gfx/interface/illustrations/decisions/decision_personal_religious.dds"
	is_shown = {
		any_secret = {
			secret_type = secret_religion
		}
	}
	effect = {
		every_secret = {
			limit = {
				secret_type = secret_religion
			}
			expose_secret = ROOT
			save_scope_as = the_secret
		}
		ROOT = {
			show_as_tooltip = {
				set_character_faith = var:false_convert
			}
		}
		every_known_secret = {
			limit = {
				secret_type = secret_religion
				NOR = {
					can_be_exposed_by = ROOT
					secret_owner = ROOT
				}
			}
			save_scope_as = other_secret
			secret_owner = {
				save_scope_as = other_secret_owner
			}
			ROOT = {
				trigger_event = {
					id = secret_religion_event.0010
					days = { 1 7 }
				}
			}
		}

	}

	title = secret_religion_reveal_faith_title
	desc = secret_religion_reveal_faith_desc
	confirm_text = secret_religion_reveal_faith_button
	selection_tooltip = secret_religion_reveal_faith_tooltip
}
