
three_teachings_convert_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_three_teachings.dds"

	cooldown = { days = 3650 }

	is_shown = {
		NOT = { faith.religious_head = root }
		OR = {
			faith = faith:zenzong
			religion = religion:taoism_religion
		}
	}

	effect = {
		trigger_event = sea_religion.0001
	}

	cost = {
		piety = 100
	}

	ai_check_interval = 1

	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100

		# Weight up for traits.
		modifier = {
			add = -1000
			has_trait = zealous
		}
	}
}
