select_faith = {
	scope = character
	saved_scopes = {
		faith
	}
	effect = {
		remove_variable = selected_culture
		set_variable = {
			name = selected_faith
			value = scope:faith
		}
	}
}

clear_selected_faith = {
	scope = character
	is_valid = {
		exists = var:selected_faith
	}
	effect = {
		remove_variable = selected_faith
	}
}

select_culture = {
	scope = character
	saved_scopes = {
		culture
	}
	effect = {
		remove_variable = selected_faith
		set_variable = {
			name = selected_culture
			value = scope:culture
		}
	}
}

clear_selected_culture = {
	scope = character
	is_valid = {
		exists = var:selected_culture
	}
	effect = {
		remove_variable = selected_culture
	}
}

clear_selected_scope = {
	scope = character
	is_valid = {
		OR = {
			exists = var:selected_culture
			exists = var:selected_faith
		}
	}
	is_shown = {
		exists = var:selected_culture
	}
	effect = {
		remove_variable = selected_culture
		remove_variable = selected_faith
	}
}
