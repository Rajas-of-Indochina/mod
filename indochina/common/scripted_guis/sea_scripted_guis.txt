select_faith = {
	scope = character
	saved_scopes = {
		faith
	}
	effect = {
		remove_variable = selected_culture
		set_variable = {
			name = selected_faith
			value = scope:faith
		}
	}
}

clear_selected_faith = {
	scope = character
	is_valid = {
		exists = var:selected_faith
	}
	effect = {
		remove_variable = selected_faith
	}
}

select_culture = {
	scope = character
	saved_scopes = {
		culture
	}
	effect = {
		remove_variable = selected_faith
		set_variable = {
			name = selected_culture
			value = scope:culture
		}
	}
}

clear_selected_culture = {
	scope = character
	is_valid = {
		exists = var:selected_culture
	}
	effect = {
		remove_variable = selected_culture
	}
}

clear_selected_scope = {
	scope = character
	is_valid = {
		OR = {
			exists = var:selected_culture
			exists = var:selected_faith
		}
	}
	is_shown = {
		exists = var:selected_culture
	}
	effect = {
		remove_variable = selected_culture
		remove_variable = selected_faith
	}
}

is_mandala_sgui = {
	scope = character
	is_shown = {
		has_government = mandala_government
	}
}

mandala_create_title = {
	scope = character
	saved_scopes = {
		title
	}
	is_valid = {
		OR = {
			AND = {
				has_government = mandala_government
				trigger_if = {
					limit = { scope:title = { tier = tier_duchy } }
					piety >= mandala_cost_Duchy
				}
				trigger_else_if = {
					limit = { scope:title = { tier = tier_kingdom } }
					piety >= mandala_cost_Kingdom
				}
				trigger_else_if = {
					limit = { scope:title = { tier = tier_empire } }
					piety >= mandala_cost_Empire
				}
				trigger_else = { always = yes }
			}
			NOT = { has_government = mandala_government }
		}
	}
}
