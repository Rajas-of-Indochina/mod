
sea_tributary_map = {
    scope = character
    effect = {
        set_global_variable = {
            name = custom_map_tributary_target
            value = root
        }
        set_global_variable = {
            name = custom_map_mode_changing
            value = 1
        }
        set_global_variable = {
            name = custom_map_mode
            value = flag:sea_tributary_map
        }
        sea_e_tributary_map = yes
        remove_global_variable = custom_map_mode_changing
    }
    is_shown = {
        exists = global_var:siro_g_initialize_complete
        global_var:custom_map_mode = flag:sea_tributary_map
    }
    is_valid = {
        NOT = { exists = global_var:custom_map_mode_changing }
        exists = global_var:siro_g_initialize_complete
    }
}
